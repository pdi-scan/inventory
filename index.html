function doGet(e) { const params = (e && e.parameter) ? e.parameter : {}; const itemId = (params.itemId || "").toString().trim(); const qty = Number(params.qty || 1); const projectId = (params.projectId || "").toString().trim(); const user = (params.user || "").toString().trim(); if (!itemId) { return output({ success: false, error: "Missing itemId", received: params }); } const result = deductStock(itemId, qty, projectId, user); return output(result); } // ============================ // STOCK DEDUCTION // ============================ function deductStock(itemId, qty, projectId, user) { const sheet = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName("INVENTORY"); const data = sheet.getDataRange().getValues(); const headers = data[0]; const idCol = headers.indexOf("Item ID"); const onHandCol = headers.indexOf("On Hand"); const cleanItem = String(itemId).trim(); for (let i = 1; i < data.length; i++) { const rowItem = String(data[i][idCol]).trim(); if (rowItem === cleanItem) { let current = Number(data[i][onHandCol]) || 0; let updated = current - qty; if (updated < 0) updated = 0; sheet.getRange(i + 1, onHandCol + 1).setValue(updated); logScan(cleanItem, qty, updated, projectId, user); return { success: true, itemId: cleanItem, remaining: updated, projectId: projectId, user: user }; } } return { success: false, error: "Item not found", itemId: cleanItem }; } // ============================ // SCAN LOG (MATCHES YOUR SHEET EXACTLY) // Timestamp | Project ID | Item ID | Qty | Remaining | Device/user // ============================ function logScan(itemId, qty, remaining, projectId, user) { const sheet = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName("SCAN_LOG"); const safeProject = projectId && projectId.trim() ? projectId : "NO_PROJECT"; const safeUser = user && user.trim() ? user : "UNKNOWN"; sheet.appendRow([ new Date(), // Timestamp (A) safeProject, // Project ID (B) itemId, // Item ID (C) qty, // Qty (D) remaining, // Remaining (E) safeUser // Device/user (F) ]); } // ============================ // RESPONSE HELPER // ============================ function output(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
